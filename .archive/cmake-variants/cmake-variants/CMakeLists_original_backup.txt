## Library Target

add_library(libgemma ${SOURCES})
set_property(TARGET libgemma PROPERTY CXX_STANDARD 20)
set_target_properties(libgemma PROPERTIES PREFIX "")
set_property(TARGET libgemma PROPERTY POSITION_INDEPENDENT_CODE ON)
target_include_directories(libgemma PUBLIC ./)
target_link_libraries(libgemma hwy hwy_contrib sentencepiece-static)
target_include_directories(libgemma PUBLIC ${sentencepiece_SOURCE_DIR})
target_compile_definitions(libgemma PRIVATE $<$<PLATFORM_ID:Windows>:_CRT_SECURE_NO_WARNINGS NOMINMAX>)
target_compile_options(libgemma PRIVATE $<$<AND:$<PLATFORM_ID:Windows>,$<CXX_COMPILER_ID:Clang>>:-Wno-deprecated-declarations>)

# Intel oneAPI Integration Configuration
if(GEMMA_USE_INTEL_MKL OR GEMMA_USE_INTEL_SIMD OR GEMMA_USE_INTEL_IPP OR GEMMA_USE_SYCL_BACKEND)
    message(STATUS "=== Intel oneAPI Integration Enabled ===")

    # Intel MKL Configuration
    if(GEMMA_USE_INTEL_MKL)
        find_package(MKL QUIET)
        if(MKL_FOUND OR EXISTS "$ENV{MKLROOT}")
            message(STATUS "Intel MKL found - enabling optimized BLAS operations")
            target_compile_definitions(libgemma PRIVATE USE_INTEL_MKL)
            target_compile_definitions(libgemma PRIVATE MKL_ILP64)
            target_compile_definitions(libgemma PRIVATE EIGEN_USE_MKL_ALL)

            if(CMAKE_CXX_COMPILER_ID MATCHES "IntelLLVM|Intel")
                target_compile_options(libgemma PRIVATE -mkl=parallel)
                target_link_options(libgemma PRIVATE -mkl=parallel)
            else()
                # Manual MKL linking for non-Intel compilers
                if(WIN32)
                    target_link_libraries(libgemma PRIVATE mkl_intel_ilp64 mkl_intel_thread mkl_core)
                else()
                    target_link_libraries(libgemma PRIVATE mkl_intel_ilp64 mkl_gnu_thread mkl_core)
                endif()
            endif()
        else()
            message(WARNING "Intel MKL requested but not found")
        endif()
    endif()

    # Intel SIMD Configuration
    if(GEMMA_USE_INTEL_SIMD)
        message(STATUS "Intel SIMD intrinsics enabled - replacing Highway scalar fallbacks")
        target_compile_definitions(libgemma PRIVATE INTEL_OPTIMIZED_BUILD)
        target_compile_definitions(libgemma PRIVATE USE_INTEL_SIMD)
        target_include_directories(libgemma PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/ops)
    endif()

    # Intel IPP Configuration
    if(GEMMA_USE_INTEL_IPP)
        if(EXISTS "$ENV{IPPROOT}")
            message(STATUS "Intel IPP found - enabling signal processing optimizations")
            target_compile_definitions(libgemma PRIVATE USE_INTEL_IPP)
            target_compile_definitions(libgemma PRIVATE IPP_STATIC_LIBS)

            if(CMAKE_CXX_COMPILER_ID MATCHES "IntelLLVM|Intel")
                target_compile_options(libgemma PRIVATE -ipp=parallel)
                target_link_options(libgemma PRIVATE -ipp=parallel)
            endif()
        else()
            message(WARNING "Intel IPP requested but not found")
        endif()
    endif()

    # SYCL Backend Configuration
    if(GEMMA_USE_SYCL_BACKEND)
        find_package(IntelSYCL QUIET)
        if(IntelSYCL_FOUND OR CMAKE_CXX_COMPILER_ID MATCHES "IntelLLVM")
            message(STATUS "Intel SYCL found - enabling cross-platform GPU support")
            target_compile_definitions(libgemma PRIVATE USE_SYCL_BACKEND)
            target_compile_options(libgemma PRIVATE -fsycl)
            target_link_options(libgemma PRIVATE -fsycl)
        else()
            message(WARNING "Intel SYCL requested but not found")
        endif()
    endif()

    message(STATUS "=== Intel oneAPI Integration Complete ===")
endif()

foreach (TESTFILE IN LISTS GEMMA_TEST_FILES)
  # The TESTNAME is the name without the extension or directory.
  get_filename_component(TESTNAME ${TESTFILE} NAME_WE)
  add_executable(${TESTNAME} ${TESTFILE})
  # Test all targets, not just the best/baseline. This changes the default
  # policy to all-attainable; note that setting -DHWY_COMPILE_* directly can
  # cause compile errors because only one may be set, and other CMakeLists.txt
  # that include us may set them.
  target_compile_options(${TESTNAME} PRIVATE -DHWY_IS_TEST=1)

  target_link_libraries(${TESTNAME} PRIVATE libgemma GTest::Main hwy hwy_contrib hwy_test)

  gtest_discover_tests(${TESTNAME})
endforeach ()
